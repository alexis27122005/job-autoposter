        <input type="email" id="email" placeholder="Email">
        <input type="password" id="password" placeholder="Contraseña">
        <button onclick="registrar()">Registrarse</button>
        <button onclick="login()" style="background: #28a745;">Iniciar Sesión</button>
    </div>

    <div id="panelUsuario" style="display:none;">
        <div class="card">
            <h2>📄 Configuración</h2>
            <input type="file" id="cv" accept=".pdf,.docx">
            <input type="text" id="pais" placeholder="País de residencia (ej: ar)">
            <input type="time" id="hora" value="09:00">
            <button onclick="guardarPreferencias()">Guardar Preferencias</button>
            <button onclick="modoPrueba()" style="background: #ffc107; color:#333;">▶️ Probar ahora</button>
        </div>

        <div class="card">
            <h2>📋 Historial de Postulaciones</h2>
            <div id="historial"></div>
        </div>
    </div>

    <script>
        let token = localStorage.getItem('token');
        const API_URL = window.location.origin + '/api';

        if(token) {
            document.querySelector('.card:first-child').style.display = 'none';
            document.getElementById('panelUsuario').style.display = 'block';
            cargarHistorial();
        }

        async function registrar() {
            const response = await fetch(API_URL + '/registro/', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    username: document.getElementById('username').value,
                    email: document.getElementById('email').value,
                    password: document.getElementById('password').value
                })
            });
            if(response.ok) alert('Registro exitoso. Ahora inicia sesión');
            else alert('Error en registro');
        }

        async function login() {
            const response = await fetch(API_URL + '/login/', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    username: document.getElementById('username').value,
                    password: document.getElementById('password').value
                })
            });
            if(response.ok) {
                const data = await response.json();
                localStorage.setItem('token', data.token);
                location.reload();
            } else alert('Error en login');
        }

        async function guardarPreferencias() {
            const formData = new FormData();
            const cvFile = document.getElementById('cv').files[0];
            if(cvFile) formData.append('cv', cvFile);
            formData.append('pais', document.getElementById('pais').value);
            formData.append('hora_postulacion', document.getElementById('hora').value);
            
            const response = await fetch(API_URL + '/perfil/', {
                method: 'PUT',
                headers: {'Authorization': `Token ${localStorage.getItem('token')}`},
                body: formData
            });
            if(response.ok) alert('Preferencias guardadas');
            else alert('Error');
        }

        async function modoPrueba() {
            const response = await fetch(API_URL + '/probar/', {
                method: 'POST',
                headers: {'Authorization': `Token ${localStorage.getItem('token')}`}
            });
            alert('Prueba iniciada. Espera unos segundos y recarga el historial');
            setTimeout(cargarHistorial, 5000);
        }

        async function cargarHistorial() {
            const response = await fetch(API_URL + '/historial/', {
                headers: {'Authorization': `Token ${localStorage.getItem('token')}`}
            });
            const data = await response.json();
            const div = document.getElementById('historial');
            div.innerHTML = '';
            data.forEach(p => {
                div.innerHTML += `
                    <div class="postulacion">
                        <strong>${p.titulo}</strong> - ${p.empresa}<br>
                        <small>${new Date(p.fecha_postulacion).toLocaleDateString()}</small>
                    </div>
                `;
            });
        }
    </script>
